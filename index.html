模板规则定义

/**
 *
 * [(n)][remark]
 *
 *  0.Replace       Default : true
 *                      If false: no replacement occurs
 *
 *  1.Sign          Default : true
 *                      Level:low
 *                      If n =  2 => start -> abc -> abc -> end
 *                      If n = -2 => start -> aa -> bb -> cc -> end
 *                      If false => n = |n|
 *
 *  2.Serial        Default : true
 *                      Level:high
 *                      If n = 2 => start -> abc -> abc -> end
 *                      If false => n = -|n| ;
 *                      serial_level > sign_level
 *
 *  3.Others        Unused
 *
**/
